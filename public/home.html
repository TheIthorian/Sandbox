Button ripple effect:

For this to work, the button must have "position:relative" and "overflow:hidden". More info can be found here.

Ripple effect 
CSS:
span.ripple {
    position: absolute; /* The absolute position we mentioned earlier */
    border-radius: 50%;
    transform: scale(0);
    animation: ripple 600ms linear;
    background-color: rgba(255, 255, 255, 0.7);
    z-index: 10;
}
@keyframes ripple {
    to {
        transform: scale(4);
        opacity: 0;
    }
}


Javascript:
function createRipple(e) {
    let button = e.currentTarget;
    
    const circle = document.createElement("span");
    const diameter = Math.max(button.clientWidth, button.clientHeight);
    const radius = diameter / 2;

    circle.style.width = circle.style.height = `${diameter}px`;
    circle.style.left = `${e.pageX - (button.offsetLeft + radius)}px`;
    circle.style.top = `${e.pageY - (button.offsetTop + radius)}px`;
    circle.classList.add("ripple"); 

    let ripple = document.getElementsByClassName("ripple")[0];

    if (ripple) {
        console.log("Removed")
        ripple.remove();
    }

    button.appendChild(circle);
}
